File: utils.c
This file contains utility functions for common mathematical operations, such as calculating the
absolute value, and finding the maximum and minimum of two values.

Functions:

1. absolute(float n):
	- Purpose: Returns the absolute value of a floating-point number.
	- Parameters:
		- float n: The number to find the absolute value of.
	- Returns: The absolute value of the number.
	- Code Documentation:

	float absolute(float n) {
		if (n < 0)
			return (-n);	// Return the negation of n if it is negative
		else
			return (n);	// Return n if it is non-negative
	}


2. get_max(float a, float b):
	- Purpose: Returns the maximum of two floating-point numbers.
	- Parameters:
			- float a: The first number.
			- float b: The second number.
	- Returns: The maximum of the two numbers.
	- Code Documentation:

	float get_max(float a, float b) {
		if (a > b)
			return (a);	// Return a if it is greater than b
		else
			return (b);	// Return b if it is greater than or equal to a
	}


3. get_min(float a, float b):
	- Purpose: Returns the minimum of two floating-point numbers.
	- Parameters:
			- float a: The first number.
			- float b: The second number.
	- Returns: The minimum of the two numbers.
	- Code Documentation:

	float get_min(float a, float b) {
		if (a < b)
			return (a);	// Return a if it is less than b
		else
			return (b);	// Return b if it is less than or equal to a
	}
